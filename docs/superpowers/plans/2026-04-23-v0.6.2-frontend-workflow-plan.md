```typescript
  outline: null,

  writtenChapters: [],

  })),

  incrementRewriteCount: () => set((state) => ({ rewriteCount: state.rewriteCount + 1 })),
  resetRewriteCount: () => set({ rewriteCount: 0 }),
  
  // ========== 工作流控制 Actions ==========
  setIsRunning: (isRunning) => set({ isRunning }),
  setWaitingForConfirmation: (waiting, type) => set({ 
    waitingForConfirmation: waiting,
    confirmationType: type || null
  }),
  clearWaitingForConfirmation: () => set({
    waitingForConfirmation: false,
    confirmationType: null
  }),
  
  // ========== SSE Actions ==========
  appendChunk: (chunk) => set((state) => ({ 
    currentChunk: state.currentChunk + chunk 
  })),
  clearChunk: () => set({ currentChunk: '' }),
  setCurrentNode: (currentNode) => set({ currentNode }),
  
  // ========== 重置 ==========
  reset: () => set(initialState),
}))
```

```typescript
/**
 * 工作流 API 客户端
 * 处理 LangGraph 工作流相关的 API 调用
 */

import { getSessionToken } from './api'
import type { WorkflowStateResponse } from '@/types'

const API_BASE_URL = import.meta.env.VITE_API_URL || ''

/**
 * 工作流 SSE 回调接口
 */
export interface WorkflowSSECallbacks {
  onNodeStart: (node: string) => void
  onNodeDone: (node: string, state: Record<string, unknown>) => void
  onChunk: (chunk: string) => void
  onWaiting: (type: string) => void
  onDone: (stage: string) => void
  onError: (error: string) => void
}

/**
 * 启动工作流（SSE 流式）
 * @param projectId - 项目 ID
 * @param callbacks - SSE 事件回调
 * @param signal - AbortSignal 用于取消请求
 */
export async function runWorkflow(
  projectId: number,
  callbacks: WorkflowSSECallbacks,
  signal?: AbortSignal
): Promise<void> {
  const token = getSessionToken()
  if (!token) {
    callbacks.onError('未登录')
    return
  }

  const credentials = btoa(`${token}:`)
  
  try {
    const response = await fetch(`${API_BASE_URL}/api/projects/${projectId}/workflow/run`, {
      method: 'POST',
      headers: {
        'Authorization': `Basic ${credentials}`,
        'Accept': 'text/event-stream',
      },
      signal,
    })

    if (!response.ok) {
      callbacks.onError(`请求失败: ${response.status}`)
      return
    }

    const reader = response.body?.getReader()
    if (!reader) {
      callbacks.onError('无法读取响应流')
      return
    }

    const decoder = new TextDecoder()
    let buffer = ''

    while (true) {
      const { done, value } = await reader.read()
      
      if (done) break

      buffer += decoder.decode(value, { stream: true })
      
      // 解析 SSE 事件
      const events = buffer.split('\n\n')
      buffer = events.pop() || ''

      for (const event of events) {
        if (!event.trim()) continue

        const lines = event.split('\n')
        let eventType = 'message'
        let eventData = ''

        for (const line of lines) {
          if (line.startsWith('event:')) {
            eventType = line.slice(6).trim()
          } else if (line.startsWith('data:')) {
            eventData += line.slice(5)
          }
        }

        if (!eventData) continue

        try {
          const data = JSON.parse(eventData)
          
          switch (eventType) {
            case 'node_start':
              callbacks.onNodeStart(data.node || '')
              break
            case 'node_done':
              callbacks.onNodeDone(data.node || '', data.state || {})
              break
            case 'waiting':
              callbacks.onWaiting(data.type || '')
              break
            case 'done':
              callbacks.onDone(data.stage || 'complete')
              break
            case 'error':
              callbacks.onError(typeof data === 'string' ? data : data.message || '未知错误')
              break
            default:
              // chunk 数据
              if (typeof data === 'string') {
                callbacks.onChunk(data)
              }
          }
        } catch {
          // 如果不是 JSON，当作普通 chunk
          callbacks.onChunk(eventData)
        }
      }
    }
  } catch (err) {
    if (err instanceof Error && err.name === 'AbortError') {
      // 用户取消，不报错
      return
    }
    callbacks.onError(err instanceof Error ? err.message : '网络错误')
  }
}

/**
 * 确认工作流当前节点
 * @param projectId - 项目 ID
 */
export async function confirmWorkflow(projectId: number): Promise<{ status: string; state: Record<string, unknown> }> {
  const token = getSessionToken()
  if (!token) {
    throw new Error('未登录')
  }

  const credentials = btoa(`${token}:`)

    },
  })

  if (!response.ok) {
    throw new Error(`确认失败: ${response.status}`)
  }

  return response.json()
}
```

```tsx
import { useWorkflowStore } from '@/stores/workflowStore'
import { CheckCircle, Circle, Loader2, AlertCircle } from 'lucide-react'
import { cn } from '@/lib/utils'

interface WorkflowStatusProps {
  className?: string
}

const stageLabels: Record<string, string> = {
  inspiration: '灵感收集',
  outline: '大纲生成',
  chapter_outlines: '章节大纲',
  writing: '章节写作',
  review: '审核中',
  complete: '已完成',
}

const nodeLabels: Record<string, string> = {
  generate_outline: '生成大纲',
  generate_chapter_outlines: '生成章节大纲',
  generate_chapter_content: '生成章节正文',
  review_chapter: '审核章节',
  rewrite_chapter: '重写章节',
}

export function WorkflowStatus({ className }: WorkflowStatusProps) {
  const stage = useWorkflowStore((state) => state.stage)
  const currentNode = useWorkflowStore((state) => state.currentNode)
  const isRunning = useWorkflowStore((state) => state.isRunning)
  const currentChapter = useWorkflowStore((state) => state.currentChapter)
  const writtenChapters = useWorkflowStore((state) => state.writtenChapters)
  const waitingForConfirmation = useWorkflowStore((state) => state.waitingForConfirmation)
  const confirmationType = useWorkflowStore((state) => state.confirmationType)

  const confirmationLabels: Record<string, string> = {
    outline: '等待确认大纲',
    chapter_outlines: '等待确认章节大纲',
    review_failed: '审核未通过',
  }

  return (
    <div className={cn('rounded-lg border p-4 space-y-3', className)}>
      {/* 阶段标题 */}
      <div className="flex items-center justify-between">
        <h3 className="font-medium">创作进度</h3>
        {isRunning && (
          <span className="text-sm text-muted-foreground flex items-center gap-1">
            <Loader2 className="h-3 w-3 animate-spin" />
            {currentNode && nodeLabels[currentNode]}
          </span>
        )}
      </div>

      {/* 阶段进度 */}
      <div className="flex items-center gap-2">
        {['inspiration', 'outline', 'chapter_outlines', 'writing', 'complete'].map((s, i) => (
          <div key={s} className="flex items-center">
            <div
              className={cn(
                'flex items-center justify-center w-6 h-6 rounded-full text-xs',
                stage === s && 'bg-primary text-primary-foreground',
                ['outline', 'chapter_outlines', 'writing', 'complete'].indexOf(stage) > i && 'bg-green-500 text-white',
                ['outline', 'chapter_outlines', 'writing', 'complete'].indexOf(stage) < i && 'bg-muted text-muted-foreground'
              )}
            >
              {['outline', 'chapter_outlines', 'writing', 'complete'].indexOf(stage) > i ? (
                <CheckCircle className="h-4 w-4" />
              ) : (
                i + 1
              )}
            </div>
            {i < 4 && (
              <div
                className={cn(
                  'w-8 h-0.5',
                  ['outline', 'chapter_outlines', 'writing', 'complete'].indexOf(stage) > i
                    ? 'bg-green-500'
                    : 'bg-muted'
                )}
              />
            )}
          </div>
        ))}
      </div>

      {/* 当前阶段 */}
      <div className="text-sm">
        <span className="text-muted-foreground">当前阶段：</span>
        <span className="font-medium">{stageLabels[stage] || stage}</span>
      </div>

      {/* 章节进度 */}
      {stage === 'writing' && writtenChapters.length > 0 && (
        <div className="text-sm">
          <span className="text-muted-foreground">章节进度：</span>
          <span className="font-medium">
            已完成 {writtenChapters.length} 章
          </span>
        </div>
      )}

      {/* 等待确认 */}
      {waitingForConfirmation && confirmationType && (
        <div className="flex items-center gap-2 p-2 bg-yellow-50 text-yellow-800 rounded-md text-sm">
          <AlertCircle className="h-4 w-4" />
          <span>{confirmationLabels[confirmationType]}</span>
        </div>
      )}
    </div>
  )
}
```

```tsx
  onViewChapters: () => void
  workflowState: WorkflowStateResponse | null
  projectName: string
}

const stageLabels: Record<string, string> = {
  inspiration: '灵感收集',
  outline: '大纲生成',
  chapter_outlines: '章节大纲',
  writing: '章节写作',
  review: '审核中',
  complete: '已完成',
}

const confirmationLabels: Record<string, string> = {
  outline: '大纲待确认',
  chapter_outlines: '章节大纲待确认',
  review_failed: '审核未通过',
}

export function ResumeDialog({
  open,
  onClose,
  onResume,
  onViewChapters,
  workflowState,
  projectName,
}: ResumeDialogProps) {
  if (!workflowState) return null

  const { stage, current_chapter, written_chapters_count, waiting_for_confirmation, confirmation_type } = workflowState

  return (
    <AlertDialog open={open} onOpenChange={(isOpen) => !isOpen && onClose()}>
      <AlertDialogContent>
        <AlertDialogHeader>
          <AlertDialogTitle className="flex items-center gap-2">
            <FileText className="h-5 w-5" />
            继续创作
          </AlertDialogTitle>
          <AlertDialogDescription asChild>
            <div className="space-y-3 text-left">
              <p className="font-medium text-foreground">{projectName}</p>
              
              <div className="space-y-2 text-sm">
                <div className="flex justify-between">
                  <span className="text-muted-foreground">当前阶段</span>
                  <span className="font-medium">{stageLabels[stage] || stage}</span>
                </div>
                
                {stage === 'writing' && (
                  <div className="flex justify-between">
                    <span className="text-muted-foreground">当前章节</span>
                    <span className="font-medium">第 {current_chapter} 章</span>
                  </div>
                )}
                
                {written_chapters_count > 0 && (
                  <div className="flex justify-between">
                    <span className="text-muted-foreground">已完成章节</span>
                    <span className="font-medium">{written_chapters_count} 章</span>
                  </div>
                )}
                
                {waiting_for_confirmation && confirmation_type && (
                  <div className="flex justify-between items-center">
                    <span className="text-muted-foreground">状态</span>
                    <span className="font-medium text-yellow-600">
                      {confirmationLabels[confirmation_type] || '等待确认'}
                    </span>
                  </div>
                )}
              </div>
            </div>
          </AlertDialogDescription>
        </AlertDialogHeader>
        <AlertDialogFooter className="flex-col gap-2 sm:flex-row">
          <AlertDialogCancel>取消</AlertDialogCancel>
          {written_chapters_count > 0 && (
            <AlertDialogAction
              onClick={onViewChapters}
              className="flex items-center gap-1"
            >
              查看已完成章节
              <ChevronRight className="h-4 w-4" />
            </AlertDialogAction>
          )}
          <AlertDialogAction onClick={onResume}>
            继续创作
          </AlertDialogAction>
        </AlertDialogFooter>
      </AlertDialogContent>
    </AlertDialog>
  )
}
```

```tsx
// 导入
import { useEffect, useState } from 'react'
import { WorkflowStatus } from '@/components/project/WorkflowStatus'
import { ResumeDialog } from '@/components/project/ResumeDialog'
import { useWorkflowStore } from '@/stores/workflowStore'
import { getWorkflowState } from '@/lib/workflowApi'
import type { WorkflowStateResponse } from '@/types'

// 在组件内
const [showResumeDialog, setShowResumeDialog] = useState(false)
const [workflowState, setWorkflowState] = useState<WorkflowStateResponse | null>(null)

const projectId = useProjectStore((state) => state.currentProject?.id)
const projectName = useProjectStore((state) => state.currentProject?.name) || ''

// 获取工作流状态
useEffect(() => {
  if (!projectId) return
  
  getWorkflowState(projectId)
    .then((state) => {
      setWorkflowState(state)
      if (state.has_checkpoint && state.stage !== 'complete') {
        setShowResumeDialog(true)
      }
    })
    .catch(console.error)
}, [projectId])

      open={showResumeDialog}
      onClose={() => setShowResumeDialog(false)}
      onResume={() => {
        setShowResumeDialog(false)
        // 启动工作流
      }}
      onViewChapters={() => {
        setShowResumeDialog(false)
        // 导航到章节列表
      }}
      workflowState={workflowState}
```
